manual set up with nginx, gunicorn and supervisor:

sudo apt install python3-pip python3-venv nginx supervisor git
git clone git@bitbucket.org:DDoneux/summonapidemo.git
cd summonapidemo
mkdir logs
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt
# add summon api access id and key to .env
# modify summonapidemo.conf and virtual.conf to use host IP
deactivate
sudo cp summonapidemo.conf /etc/supervisor/conf.d/
sudo supervisorctl reread
sudo service supervisor restart
sudo supervisor status
sudo cp virtual.conf /etc/nginx/conf.d/
sudo nginx -t
sudo service nginx restart


docker:

sudo apt install git python3-pip docker.io
git clone git@bitbucket.org:DDoneux/summonapidemo.git
cd summonapidemo
sudo docker build -t summonapidemo .
# add summon api access id and key to .env
sudo docker run -p 80:80 --env-file=.env -t summonapidemo
